					" SELECT 		R.DOCUMENT_NUMBER as documentNumber," +

					"       		N.TEL_NBR_NPA || N.TEL_NBR_NXX || MAX (N.TEL_NBR_LINE_RANGE) as rangeFinal " +

					"       		ASAP.SERV_REQ R " + 
					"WHERE 		R.DOCUMENT_NUMBER = N.DOCUMENT_NUMBER " + 
					"AND 		R.PON = ? " + 
					"AND 		R.TYPE_OF_SR = 'SO' " + 
					"GROUP BY 	R.DOCUMENT_NUMBER, " +
